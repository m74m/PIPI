#include <stdio.h>


int main(int argc, char *argv[])
{
  int p[1000], i , cl=0, djeljivih, j;
  for (i=0; i<1000; i++){
      scanf ("%d", &p[cl++]);
      if (p[cl-1]==0) {cl-=1; break;}
      }
  for (i=0; i<cl; i++) {
      djeljivih=0;
      for (j=0; j<cl;j++) {
          if (i==j) continue;
          else {
                 if (p[i]%p[j]==0) djeljivih++;
                 }
                 }
      printf ("%d: %d\n", p[i], djeljivih);
}
  
  system("PAUSE");	
  return 0;
}

#include <stdio.h>
#define MAX 60
int main(int argc, char *argv[])
{
  int m1[MAX][MAX], m2[MAX][MAX], i,j, m, n;
  do {
      scanf ("%d %d", &m, &n);
      } while (m<1 || n<1 || m>MAX || n>MAX);
  for (i=0; i<m; i++) {
      for (j=0; j<n; j++) {
          scanf ("%d", &m1[i][j]);
          m2[j][i]= m1 [i][j];
          }
          }
  for (i=0; i<n; i++) {
      for (j=0; j<m; j++) {
          printf ("%d", m2 [i][j]);
          }
      printf ("\n");
      }
  system("PAUSE");	
  return 0;
}
